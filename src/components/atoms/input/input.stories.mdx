### With Value and Listenner (controll use):

<Canvas>
  <Story name="Value and Listenner">
    {{
        components: { HInput },
        template: `
        <div>
          <h-input type="text" :value="usingNormalData" @input="handleInput" placeholder="default" />
          {{ usingNormalData }}
          <h-input type="text" v-model="usingVmodel" placeholder="v-model" />
          {{ usingVmodel }}
        </div>
        `,
        data(){
          return {
            usingNormalData: '',
            usingVmodel: '',
          };
        },
        methods: {
          handleInput(value) {
            this.usingNormalData = value;
          },
        },
    }}
  </Story>
</Canvas>
