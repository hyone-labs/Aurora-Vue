### With listenners use:

<Canvas>
  <Story name="With listenners use">
    {{
        components: { ASelect },
        data() {
          return {
            selected: 'blah',
          };
        },
        template: `
        <div style="height: 40vh;">
          <a-select
            @select="selected = $event"
            :value="selected"
            :options="['example1', 'Example 1']"
          />
          <a-select
            @input="selected = $event"
            :value="selected"
            :options="['example1', 'Example 1']"
          />
          <a-select
            @change="selected = $event"
            :value="selected"
            :options="['example1', 'Example 1']"
          />
          {{ selected }}
        </div>
        `,
    }}
  </Story>
</Canvas>
