### Sizes:

<Canvas>
  <Story name="Sizes">
    {{
      components: { HModal, HButton },
      data() {
        return { active: false, size: 'default' };
      },
      methods: {
        handleClick(size) {
          this.active = true;
          this.size = size;
        },
      },
      template: `<div>
        <h-button @click="handleClick('default')">default</h-button>
        <h-button @click="handleClick('small')">small</h-button>
        <h-button @click="handleClick('medium')">medium</h-button>
        <h-button @click="handleClick('large')">large</h-button>
        <h-modal :active="active" @deactivating="active = false" :size="size">HModal</h-modal>
      </div>`,
    }}
  </Story>
</Canvas>
